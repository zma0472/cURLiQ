  ALTER SESSION SET attr = value [ , attr = value ] ... ;

  DISPLAY SESSION [ attr [ , attr ] ... ] ;

  END SESSION ;

  ALTER SYSTEM SET attr = value [ , attr = value ] ... ;

  DISPLAY SYSTEM [ attr [ , attr ] ... ] ;

  CONNECT ;

  DISCONNECT ;

  CREATE QUEUE name ( [ attr = value [ , attr = value ] ... ] ) ;

  ALTER QUEUE name SET attr = value [ , attr = value ] ... ;

  ALTER QUEUE name ADD TARGET name [ , name ] ... ;

  ALTER QUEUE name REMOVE TARGET name < [ , name ] ... | ALL > ;

  ONLINE QUEUE name ;

  OFFLINE QUEUE name ;

  DISPLAY QUEUE [ attr [ , attr ] ... ] ;

  DROP QUEUE name [ CASCADE ] ;
  #  CASCADE drops all targets whose only queue is being dropped.



  CREATE TARGET name ( [ attr = value [ , attr = value ] ... ] ) ;

  ALTER TARGET name SET attr = value [ , attr = value ] ... ;

  ONLINE TARGET target ;

  OFFLINE TARGET target ; 

  DISPLAY TARGET [ attribute ] ;

  DROP TARGET target [ CASCADE ] ;
  #  CASCADE drops all queues whose only target is being dropped.



  CREATE QUEUE GROUP name ;

  ALTER QUEUE GROUP name SET attribute = value ;

  ALTER QUEUE GROUP name ADD QUEUE queue_name ;

  ALTER QUEUE GROUP name REMOVE QUEUE < queue_name | ALL > ;

  ONLINE QUEUE GROUP name ;

  OFFLINE QUEUE GROUP name ;

  DISPLAY QUEUE GROUP [ attribute ] ;

  DROP QUEUE GROUP name [ CASCADE ] ;
  #  CASCADE drops all queues contained exclusively in the queue group being
  #  dropped. 



  CREATE TARGET GROUP name ;

  ALTER TARGET GROUP name SET attribute = value ;

  ALTER TARGET GROUP name ADD TARGET target ;

  ALTER TARGET GROUP name REMOVE TARGET < target_name | ALL > ;

  ONLINE TARGET GROUP name ;

  OFFLINE TARGET GROUP name ;

  DISPLAY TARGET GROUP [ attribute ] ;

  DROP TARGET GROUP name [ CASCADE ] ;
  #  CASCADE drops all targets contained exclusively in the target group being
  #  dropped.
