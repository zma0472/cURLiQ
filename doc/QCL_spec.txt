#
#  Preliminary definitions:
#
#  QUEUE -- A named object consisting of queue attributes, a single source
#           directory, and zero or more TARGETS.
#
#  QUEUE GROUP -- A named object consisting of queue group attributes, queue
#                 attributes, and an unordered set of QUEUEs.
#
#  TARGET -- A named object consisting of target attributes, a single URL, and
#            a collection of cURL parameters.
#
#  TARGET GROUP -- A named object consisting of target group attributes, a
#                  collection of cURL parameters, and an unordered set of
#                  TARGETs.
#
#  TRIGGER -- A named object consisting of trigger attributes, an event, and an
#             action.
#
#  TRIGGER GROUP -- A named object consisting of trigger group attributes,
#                   trigger attributes, and an unordered collection of
#                   TRIGGERS.
#
#  BROKER -- A named object consisting of broker attributes, QUEUEs, QUEUE
#            GROUPs, TARGETs, TARGET GROUPs, TRIGGERs, and TRIGGER GROUPs.
#

#
# SESSION commands.
#
  START SESSION ;

  ALTER SESSION SET attr = value [ , attr = value ] ... ;

  DISPLAY SESSION [ attr [ , attr ] ... ] ;

  END SESSION ;

#
# BROKER commands.
#
  CREATE BROKER broker ( [ attr = value [ , attr = value ] ... ] ) ;

  ALTER BROKER broker SET attr = value [ , attr = value ] ... ;

  ALTER BROKER broker ADD TARGET target [ , target ] ... ;

  ALTER BROKER broker ADD QUEUE queue [ , queue ] ... ;

  ALTER BROKER broker REMOVE TARGET name [ [ , name ] ... | ALL ] ;

  ALTER BROKER broker REMOVE QUEUE queue [ [ , queue ] ... | ALL ] ;

  ALTER BROKER broker SHARE TARGET name [ , name ... | ALL ]  WITH
                                                 broker [ , broker ... | ALL ] ;

  ONLINE BROKER broker ;

  OFFLINE BROKER broker ;

  DISPLAY BROKER broker [ attr [ , attr ] ... ] ;

  DROP BROKER broker [ CASCADE ] ;
  #  CASCADE drops all unshared TARGETs, QUEUEs, and TRIGGERs belonging to the
  #  BROKER being dropped.  In this case the orphaned objects remain.

#
# QUEUE commands.
#
  CREATE QUEUE name ( [ attr = value [ , attr = value ] ... ] ) ;

  ALTER QUEUE name SET attr = value [ , attr = value ] ... ;

  ALTER QUEUE name ADD TARGET name [ , name ] ... ;

  ALTER QUEUE name REMOVE TARGET name < [ , name ] ... | ALL > ;

  ONLINE QUEUE name ;

  OFFLINE QUEUE name ;

  DISPLAY QUEUE name [ attr [ , attr ] ... ] ;

  DROP QUEUE name [ CASCADE ] ;
  #  CASCADE drops all targets whose only queue is being dropped.


  CREATE QUEUE GROUP name ;

  ALTER QUEUE GROUP name SET attribute = value ;

  ALTER QUEUE GROUP name ADD QUEUE queue_name ;

  ALTER QUEUE GROUP name REMOVE QUEUE < queue_name | ALL > ;

  ONLINE QUEUE GROUP name ;

  OFFLINE QUEUE GROUP name ;

  DISPLAY QUEUE GROUP name [ attribute ] ;

  DROP QUEUE GROUP name [ CASCADE ] ;
  #  CASCADE drops all queues contained exclusively in the group being dropped. 


  CREATE TARGET name ( [ attr = value [ , attr = value ] ... ] ) ;

  ALTER TARGET name SET attr = value [ , attr = value ] ... ;

  ONLINE TARGET name ;

  OFFLINE TARGET name ; 

  DISPLAY TARGET name [ attribute ] ;

  DROP TARGET target [ CASCADE ] ;
  #  CASCADE drops all queues whose only target is being dropped.


  CREATE TARGET GROUP name ;

  ALTER TARGET GROUP name SET attribute = value ;

  ALTER TARGET GROUP name ADD TARGET target ;

  ALTER TARGET GROUP name REMOVE TARGET < target_name | ALL > ;

  ONLINE TARGET GROUP name ;

  OFFLINE TARGET GROUP name ;

  DISPLAY TARGET GROUP name [ attribute ] ;

  DROP TARGET GROUP name [ CASCADE ] ;
  #  CASCADE drops all targets contained exclusively in the group being
  #  dropped. 


  CREATE TRIGGER name ( [ attr = value [ , attr = value ] ... ] ) ;

  ALTER TRIGGER name SET attr = value [ , attr = value ] ... ;

  ONLINE TRIGGER name ;

  OFFLINE TRIGGER name ; 

  DISPLAY TRIGGER name [ attribute ] ;

  DROP TRIGGER name [ CASCADE ] ;
  #  CASCADE drops all queues whose only target is being dropped.

  CREATE TRIGGER GROUP name ;

  ALTER TRIGGER GROUP name SET attribute = value ;

  ALTER TRIGGER GROUP name ADD TRIGGER target ;

  ALTER TRIGGER GROUP name REMOVE TRIGGER < trigger_name | ALL > ;

  ONLINE TRIGGER GROUP name ;

  OFFLINE TRIGGER GROUP name ;

  DISPLAY TRIGGER GROUP name [ attribute ] ;

  DROP TRIGGER GROUP name [ CASCADE ] ;
  #  CASCADE drops all triggers contained exclusively in the trigger group
  #  being dropped.
